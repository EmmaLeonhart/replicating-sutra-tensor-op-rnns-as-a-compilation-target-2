% paper.tex — Sutra paper, NeurIPS 2026 submission wrapper.
%
% The substantive paper content lives in paper/paper.md. The
% paper-pdf.yml workflow runs `pandoc paper.md -t latex -o
% paper.tex.body` before pdflatex, so the entire body of the
% markdown lands here at \input time. paper.tex itself is the
% NeurIPS-2026 wrapper that supplies the documentclass, style
% file options, packages, anonymization switch, title, and
% author block.
%
% Two builds from the same source via the \ifanon switch:
%   pdflatex paper.tex                                     -> named (preprint, no line numbers)
%   pdflatex -usepretex='\newif\ifanon\anontrue' paper.tex -> anonymized submission (line numbers)
%
% Build chain:
%   paper-pdf.yml -> pandoc paper.md -t latex -o paper.tex.body
%                 -> latexmk -pdf paper.tex                      (named/preprint)
%                 -> latexmk -pdf -usepretex='...' paper.tex     (anonymized submission)

\documentclass{article}

% Anonymization switch — set to true via -usepretex for the
% submission build. Default is named/preprint.
\makeatletter
\@ifundefined{ifanon}{\newif\ifanon}{}
\makeatother

% Load the official NeurIPS 2026 style file.
%   - default (no option): anonymous + line numbers, for submission
%   - [preprint]: non-anonymous, no line numbers, for arXiv/clawRxiv
% nonatbib disables the natbib package because the paper uses an
% unnumbered bullet-list References section, not natbib citations.
\ifanon
  \usepackage[nonatbib]{neurips_2026}
\else
  \usepackage[preprint,nonatbib]{neurips_2026}
\fi

% pdflatex Unicode handling.
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}

% Standard packages we use throughout the body.
\usepackage{hyperref}
\usepackage{url}
\usepackage{xcolor}
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{booktabs}
\usepackage{longtable}
\usepackage{array}
% pandoc's longtable column-width syntax uses arithmetic on
% \linewidth ("(\linewidth - 8\tabcolsep) * \real{0.6098}") which
% needs the calc package and the \real macro from pandoc's default
% preamble. Without these, every multi-column table renders the
% column-width fractions as literal text.
\usepackage{calc}
\providecommand{\real}[1]{#1}
\usepackage{caption}
\usepackage{enumitem}
\usepackage{verbatim}
\usepackage{tikz}
\usetikzlibrary{positioning,arrows.meta,shapes.geometric,calc,fit,backgrounds}

% Map the math-style Unicode used in the body prose (≈, ×, ∈,
% superscript digits, etc.) onto the LaTeX equivalents Times
% Roman + T1 fontenc has glyphs for. Without this, pdflatex
% emits "missing character" warnings and the glyphs render as
% blanks. newunicodechar works under pdflatex when used after
% inputenc.
\usepackage{newunicodechar}
\newunicodechar{≈}{\ensuremath{\approx}}
\newunicodechar{≠}{\ensuremath{\neq}}
\newunicodechar{×}{\ensuremath{\times}}
\newunicodechar{−}{\ensuremath{-}}
\newunicodechar{∈}{\ensuremath{\in}}
\newunicodechar{‖}{\ensuremath{\Vert}}
\newunicodechar{→}{\ensuremath{\rightarrow}}
\newunicodechar{…}{\ensuremath{\ldots}}
\newunicodechar{‑}{-}
\newunicodechar{⁰}{\textsuperscript{0}}
\newunicodechar{¹}{\textsuperscript{1}}
\newunicodechar{²}{\textsuperscript{2}}
\newunicodechar{³}{\textsuperscript{3}}
\newunicodechar{⁴}{\textsuperscript{4}}
\newunicodechar{⁵}{\textsuperscript{5}}
\newunicodechar{⁶}{\textsuperscript{6}}
\newunicodechar{⁷}{\textsuperscript{7}}
\newunicodechar{⁸}{\textsuperscript{8}}
\newunicodechar{⁹}{\textsuperscript{9}}
\newunicodechar{⁻}{\textsuperscript{\ensuremath{-}}}
\newunicodechar{§}{\S}

% pandoc longtable + hyperref shim. Pandoc emits unnumbered tables
% (longtable, no caption); hyperref tries to anchor on counter 'none'
% (longtable's \LTcaptype default for uncaptioned tables). Strict
% pdflatex (e.g. arXiv's TeX Live with -interaction=batchmode +
% file-line-error) fails with "No counter 'none' defined" without
% this. The CI's latexmk -interaction=nonstopmode silently continued
% past the error and produced a PDF, hiding the issue from the
% workflow. Define the counter + the hyperref-side representation
% before they're needed.
\makeatletter
\@ifundefined{c@none}{\newcounter{none}}{}
\makeatother
\providecommand{\theHnone}{none.\arabic{none}}

% pandoc emission shims:
\providecommand{\tightlist}{\setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}
\providecommand{\pandocbounded}[1]{#1}
% pandoc's `Shaded` (background-shaded code block).
\usepackage{framed}
\definecolor{shadecolor}{RGB}{248,248,248}
\newenvironment{Shaded}{\begin{snugshade*}}{\end{snugshade*}}

\hypersetup{
  colorlinks=true,
  linkcolor=blue,
  urlcolor=blue,
  citecolor=blue,
}

\title{Sutra: Tensor-Op RNNs as a Compilation Target\\
  for Vector Symbolic Architectures}

% NeurIPS submission: \author block is rendered as "Anonymous
% Authors" automatically by neurips_2026.sty when \ifanon is set
% (no [preprint] option). For the named/preprint build, list the
% real author info.
\ifanon
  \author{Anonymous Authors}
\else
  \author{%
    Emma Leonhart\\
    \texttt{EmmaLeonhart999@gmail.com}%
  }
\fi

\begin{document}
\maketitle

% paper.tex.body`, generated by the paper-pdf.yml workflow before
% pdflatex runs. Everything below — Abstract, sections 1-6,
% references, appendices A-K — is sourced from paper.md.
\input{paper.tex.body}

\end{document}
